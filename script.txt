units real
boundary p p p
atom_style full
read_data lmp.data

bond_style harmonic
angle_style charmm
dihedral_style charmm
improper_style harmonic

kspace_style pppm 1e-4
pair_style hybrid lj/cut/coul/long 10.0

# 1  C
# 2  Fe
# 3  H
# 4  N
# 5  O
# 6  S

# Pair coefficients (unchanged)
pair_coeff 3 3 lj/cut/coul/long 0.5 2.846
pair_coeff 5 5 lj/cut/coul/long 0.2  2.85
pair_coeff 4 4 lj/cut/coul/long 0.17 3.250
pair_coeff 1 1 lj/cut/coul/long 0.12 3.30
pair_coeff 6 6 lj/cut/coul/long 0.27 3.59
pair_coeff 2 2 lj/cut/coul/long 0.013 2.32
pair_modify mix arithmetic
neigh_modify delay 1 every 1 check yes one 2000
bond_coeff 1 0 0

# Create results folder
variable results string "simulation_results"
shell rmdir ${results}
shell mkdir ${results}
shell cd ${results}

# Define ligand region and group
region ligand_region block 48.5 50.5 -43.5 -41 12.9 14.7
group ligand_group region ligand_region
set group ligand_group charge -0.5

# Set Fe charge
set type 2 charge 5  # Warning: Extremely high charge (verify if intentional)

# Define groups for Fe and charged O
group Fe_Atoms type 2
#group O_ligand type 5 region ligand_region  # Critical: Only 1 O in this region

# Verify groups contain exactly 1 atom each
variable Fe_count equal count(Fe_Atoms)
variable O_count equal count(ligand_group)
print "Fe atoms: ${Fe_count}, O in ligand: ${O_count}"

# Compute coordinates using global averages (works for single-atom groups)
compute Fe_x Fe_Atoms property/atom xs
compute Fe_y Fe_Atoms property/atom yu
compute Fe_z Fe_Atoms property/atom zu
compute O_x ligand_group property/atom xu
compute O_y ligand_group property/atom yu
compute O_z ligand_group property/atom zu

compute Fe_avg all reduce ave c_Fe_x c_Fe_y c_Fe_z
compute O_avg all reduce ave c_O_x c_O_y c_O_z

# Calculate distance
#variable dx equal c_O_x-c_Fe_x
variable dx equal c_O_avg[1]-c_Fe_avg[1]
variable dy equal c_O_avg[2]-c_Fe_avg[2]
variable dz equal c_O_avg[3]-c_Fe_avg[3]
variable distance equal v_dx#sqrt(v_dx^2+v_dy^2+v_dz^2)

# Output
dump 1 all custom 100 dmp.*.txt id type x y z mass
thermo_style custom step temp press lx ke pe etotal density v_distance
thermo 100

# Simulation
minimize 0.0 0.0 1000 10000
velocity all create 300.0 1234567 mom yes rot yes dist gaussian
fix 1 all nve
timestep 0.1
run 30000
